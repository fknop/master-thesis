    & w_{ijkl} \in \{0, 1\}, && \forall i \in W, j \in T, k \in D, l \in d^P_k \label{binary1} \\

    & m_{ij} \in \{0, 1\}, && \forall i \in M, j \in D \label{binary3} \\ 
    & z_{ij} \in \{0, 1\}, && \forall i \in Z, j \in D \label{binary4} \\
